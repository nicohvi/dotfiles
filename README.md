# Installation

    git clone https://github.com/nicohvi/dotvim ~/.vim
    ln -s ~/.vim/vimrc ~/.vimrc
    ln -s ~/.vim/gvimrc ~/.gvimrc
    cd ~/.vim
    git submodule update --init --recursive
